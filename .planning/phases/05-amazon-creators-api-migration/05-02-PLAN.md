5. **New `_search_real_api` method** (replaces PA-API version):
```python
async def _search_real_api(
    self,
    query: str,
    category: Optional[str],
    brand: Optional[str],
    min_price: Optional[float],
    max_price: Optional[float],
    limit: int,
    country_code: str,
) -> List[AffiliateProduct]:
    """
    Search products using Amazon Creators API via python-amazon-paapi library.
    OAuth2 token management is handled by the library internally.
    """
    if not self.credential_id or not self.credential_secret:
        logger.error(
            "Amazon Creators API credentials not configured. "
            "Set AMAZON_CREDENTIAL_ID and AMAZON_CREDENTIAL_SECRET in .env"
        )
        return await self._search_mock_data(
            query, category, brand, min_price, max_price, limit, country_code
        )

    country_tags = parse_associate_tags(settings.AMAZON_ASSOCIATE_TAGS)
    partner_tag = country_tags.get(country_code.upper()) or self.associate_tag

    if not partner_tag:
        logger.error("No Amazon Associate tag configured for country: " + country_code)
        return []

    country_enum = COUNTRY_MAP.get(country_code.upper(), Country.US)
    search_index = self._map_category_to_search_index(category)

    try:
        async with AsyncAmazonCreatorsApi(
            credential_id=self.credential_id,
            credential_secret=self.credential_secret,
            tag=partner_tag,
            country=country_enum,
            throttling=1.0,
        ) as api:
            results = await api.search_items(
                keywords=query,
                search_index=search_index,
                item_count=min(limit, 10),
            )

            if not results or not results.items:
                logger.warning("Empty response from Creators API, falling back to mock data")
                return await self._search_mock_data(
                    query, category, brand, min_price, max_price, limit, country_code
                )

            return self._convert_to_affiliate_products(results.items, country_code)

    except Exception as e:
        logger.error(f"Creators API request failed: {e}", exc_info=True)
        return await self._search_mock_data(
            query, category, brand, min_price, max_price, limit, country_code
        )
```

6. **New `_convert_to_affiliate_products` method** (replaces `_parse_paapi_response`):
```python
def _convert_to_affiliate_products(
    self, items: list, country_code: str
) -> List[AffiliateProduct]:
    """Convert python-amazon-paapi library response objects to AffiliateProduct list."""
    results = []

    for item in items:
        try:
            asin = item.asin

            # Title
            title = "Unknown Product"
            if item.item_info and item.item_info.title:
                title = item.item_info.title.display_value

            # Image URL (direct Amazon CDN hotlink -- AMZN-03)
            image_url = ""
            if item.images and item.images.primary and item.images.primary.large:
                image_url = item.images.primary.large.url

            # Price
            price = 0.0
            currency = "USD"
            if item.offers and item.offers.listings:
                listing = item.offers.listings[0]
                if listing.price:
                    price = listing.price.amount
                    currency = listing.price.currency

            # Reviews
            rating = None
            review_count = None
            if item.customer_reviews:
                if item.customer_reviews.star_rating:
                    rating = float(item.customer_reviews.star_rating.value)
                if item.customer_reviews.count:
                    review_count = int(item.customer_reviews.count)

            # Affiliate link
            affiliate_link = generate_amazon_affiliate_link(
                asin=asin,
                country_code=country_code,
                associate_tag=self.associate_tag,
            )

            results.append(AffiliateProduct(
                product_id=asin,
                title=title,
                price=price,
                currency=currency,
                affiliate_link=affiliate_link,
                merchant="Amazon",
                image_url=image_url,
                rating=rating,
                review_count=review_count,
                condition="new",
                availability=True,
                source_url=affiliate_link,
            ))

        except Exception as e:
            logger.warning(f"Failed to parse Creators API item: {e}")
            continue

    logger.info(f"Creators API: parsed {len(results)} products")
    return results
```
